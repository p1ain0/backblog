约瑟夫问题是个有名的问题：N个人围成一圈，从第一个开始报数，第M个将被杀掉，最后剩下一个，其余人都将被杀掉。例如N=6，M=5，被杀掉的顺序是：5，4，6，2，3，1。

代码实现：

```c++
void Joseph(int n, int m)
{
    int a[100]; //开局每个人的编号
    int k = 0;  //记录第k个出局的人
    int p = 0;  //出局人的当前位置
    for (int i = 0; i < n; i++){
        a[i] = i + 1; //编号初始化，
    }
    //循环T人
    while (n > 1){
        p = (p + m - 1) % n; //关键点，计算出圈人的位置
        cout << "第" << ++k << "个出圈的是：" << a[p] << endl;
        //重新排序
        for (int j = p + 1; j < n; j++)
            a[j - 1] = a[j]; 
        //人数减一
        n--;
        if (p == n) p = 0;

    }
    cout << "最后剩下的是" << a[p] << endl;

}
```
